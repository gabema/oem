ToolhouseUI.Components = (function(Components, Core) {

    // COMPONENT_CLASS component
    var COMPONENT_CLASSTest = Object.create(Core.Test); // call super constructor
    COMPONENT_CLASSTest.name = "Tests";
    COMPONENT_CLASSTest.testComponent = 'COMPONENT_NAME-test';

    /**
     * Test example
     *
     * @method     
     */
    COMPONENT_CLASSTest.exampleIsWorking = function(){
        COMPONENT_CLASSTest.assert('Test example is working', true, true);
    };    

    /**
     * Run tests
     */
    Core.Events.addEventListener(Core.EVENTS.DOCUMENT_READY, function(){
        COMPONENT_CLASSTest.runTestSuite([
            COMPONENT_CLASSTest.exampleIsWorking
        ]);
    });

    // exports
    Components.COMPONENT_CLASSTest = COMPONENT_CLASSTest;
    return Components;

})(ToolhouseUI.Components || {}, ToolhouseUI.Core);
